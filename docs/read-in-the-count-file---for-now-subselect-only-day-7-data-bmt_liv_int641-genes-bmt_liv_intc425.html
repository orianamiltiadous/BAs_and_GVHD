# run voom

# fit linear model
fit <- lmFit(v, designBMT_liv_int)

                     LIV_BMT-LIV_BM, levels=designBMT_liv_int)

# differential expression analysis
fit2 <- contrasts.fit(fit, cm)
fit2 <- eBayes(fit2)
fit2$groupMeans <- fit$coefficients
